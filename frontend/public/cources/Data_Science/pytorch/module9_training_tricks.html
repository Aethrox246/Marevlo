defmixup_batch(X:torch.Tensor,y:torch.Tensor,alpha:float=0.4): """
Applymixupaugmentationtoabatch.

Createsconvexcombinationsofpairsofsamples.

Returns:
X_mixed:interpolatedfeatures
y_a,y_b:originallabels

lam:

mixingcoefficient

"""

lam

=np.random.beta(alpha,alpha)ifalpha>0else1.0

=torch.randperm(len(X))

X_mixed=lam*X+(1-lam)*X[idx]

y_a

=y

y_b

=y[idx]

returnX_mixed,y_a,y_b,lam

defmixup_criterion(criterion,pred,y_a,y_b,lam):
"""Lossformixup-augmentedsamples."""
returnlam*criterion(pred,y_a)+(1-lam)*criterion(pred,y_b)

classFocalLoss(nn.Module):
"""
FocalLoss:FL(p_t)=-alpha_t*(1-p_t)^gamma*log(p_t) gamma>0reduceslossforwell-classifiedexamples.

alphabalancesclassimportance.

"""

def__init__(self,alpha:float=0.25,gamma:float=2.0, reduction:str='mean'):
super().__init__()

self.alpha

=alpha

self.gamma

=gamma

self.reduction=reduction
defforward(self,logits:torch.Tensor,targets:torch.Tensor)->torch.Tensor: ce_loss=nn.functional.cross_entropy(logits,targets,reduction='none')

=torch.exp(-ce_loss)

=self.alpha*(1-pt)**self.gamma*ce_loss

5.Self-SupervisedLearning:ContrastiveLearningIntro

Contrastive learning trains a model to bring similar samples together and push different samples

apart—withoutrequiringlabelsforeverysample.

classProjectionHead(nn.Module):
"""SmallMLPheadusedforcontrastivelearning.""" def__init__(self,in_dim,proj_dim=64):
super().__init__()
self.net=nn.Sequential(
nn.Linear(in_dim,in_dim),nn.ReLU(),
nn.Linear(in_dim,proj_dim)
)
defforward(self,x):returnself.net(x)

defnt_xent_loss(z1:torch.Tensor,z2:torch.Tensor,temperature:float=0.5):

"""
NT-Xent(NormalisedTemperature-scaledCrossEntropy)loss.

z1,z2:twoaugmentedviewsofthesamesamples(B,D).

6

Positives:(z1[i],z2[i]).Negatives:allotherpairs.

"""
B=z1.size(0)

z=torch.cat([z1,z2],dim=0)

#(2B,D)

z=nn.functional.normalize(z,dim=1)

sim

=torch.matmul(z,z.T)/temperature

#(2B,2B)

#Removeself-similarityfromdenominator
mask=torch.eye(2*B,dtype=torch.bool,device=z.device) sim.masked_fill_(mask,float('-inf'))

#Positivepairs:iwithi+Bandi+Bwithi
labels=torch.cat([torch.arange(B)+B,
torch.arange(B)],dim=0).to(z.device)

returnnn.CrossEntropyLoss()(sim,labels)
