defmixup_batch(X:torch.Tensor,y:torch.Tensor,alpha:float=0.4): """ Applymixupaugmentationtoabatch. Createsconvexcombinationsofpairsofsamples. Returns: X_mixed:interpolatedfeatures y_a,y_b:originallabels lam: mixingcoefficient """ lam =np.random.beta(alpha,alpha)ifalpha>0else1.0 idx =torch.randperm(len(X)) X_mixed=lam*X+(1-lam)*X[idx] y_a =y y_b =y[idx] returnX_mixed,y_a,y_b,lamdefmixup_criterion(criterion,pred,y_a,y_b,lam): """Lossformixup-augmentedsamples.""" returnlam*criterion(pred,y_a)+(1-lam)*criterion(pred,y_b)#Usageintrainingloop X_b=torch.randn(32,10)

4.ClassImbalance:FocalLoss+WeightedSampling

FocalLoss

Focallossdown-weightseasy,well-classifiedexamplessothemodelfocusesonhardones.

Originallydesignedforobjectdetection,it’shighlyusefulforanomalydetectionandfraud.

<python>

classFocalLoss(nn.Module): """ FocalLoss:FL(p_t)=-alpha_t*(1-p_t)^gamma*log(p_t) gamma>0reduceslossforwell-classifiedexamples. alphabalancesclassimportance. """ def__init__(self,alpha:float=0.25,gamma:float=2.0, reduction:str='mean'): super().__init__() self.alpha =alpha self.gamma =gamma self.reduction=reduction defforward(self,logits:torch.Tensor,targets:torch.Tensor)->torch.Tensor: ce_loss=nn.functional.cross_entropy(logits,targets,reduction='none') pt =torch.exp(-ce_loss) #p_t=probabilityofcorrectclass focal =self.alpha*(1-pt)**self.gamma*ce_lossifself.reduction=='mean':returnfocal.mean()
