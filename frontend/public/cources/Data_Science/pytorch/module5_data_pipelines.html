np.random.seed(99)

defgenerate_log_stream(n_events:int=5000,vocab_size:int=50, anomaly_rate:float=0.05):
"""
Simulatealogstream.

Normal:Markov-ishtransitionsbetweencommontemplates. Anomaly:randomraretemplatesinjected.

"""

events

=[]

labels

=[]

#1=anomalywindow,0=normalwindow

templates=np.arange(vocab_size)

#Buildatransitionmatrix(normalbehaviour)
trans=np.random.dirichlet(np.ones(vocab_size)*0.3,size=vocab_size)

current=0
foriinrange(n_events):
is_anomaly=np.random.rand()<anomaly_rate
ifis_anomaly:
#Raretemplate—highID
event=np.random.randint(40,vocab_size)
else:
event=np.random.choice(templates,p=trans[current]) events.append(event)
labels.append(int(is_anomaly))
current=event

returnnp.array(events),np.array(labels)
