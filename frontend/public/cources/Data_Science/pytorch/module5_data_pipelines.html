The Dataset Contract
__len__()  ·  __getitem__(idx)
Any class that implements these two methods is a valid PyTorch Dataset. __len__ returns the total number of samples. __getitem__ returns a single (input, label) pair for a given index.
Python
import torch
from torch.utils.data import Dataset, DataLoader
import numpy as np

# ── Custom Dataset ────────────────────────────────────────────────
class TabularDataset(Dataset):
    def __init__(self, X: np.ndarray, y: np.ndarray, transform=None):
        self.X = torch.tensor(X, dtype=torch.float32)
        self.y = torch.tensor(y, dtype=torch.float32)
        self.transform = transform

    def __len__(self):
        return len(self.X)

    def __getitem__(self, idx):
        x, y = self.X[idx], self.y[idx]
        if self.transform:
            x = self.transform(x)
        return x, y

# Create dummy data
X = np.random.randn(1000, 20)
y = np.random.randint(0, 2, size=(1000,))

dataset  = TabularDataset(X, y)
print(f"Dataset size: {len(dataset)}")
print(f"Sample shape: {dataset[0][0].shape}, label: {dataset[0][1]}")
Output
Dataset size: 1000
Sample shape: torch.Size([20]), label: tensor(1.)
02 — DataLoader Configuration

Serving Data Efficiently

Python
# Train/Val/Test split
from torch.utils.data import random_split

train_size = int(0.8 * len(dataset))
val_size   = len(dataset) - train_size
train_ds, val_ds = random_split(dataset, [train_size, val_size],
                                  generator=torch.Generator().manual_seed(42))

# DataLoaders
train_loader = DataLoader(
    train_ds,
    batch_size=64,
    shuffle=True,          # Shuffle training data every epoch
    num_workers=4,         # Background workers for loading
    pin_memory=True,       # Faster GPU transfer (use if GPU available)
    drop_last=True,        # Drop incomplete final batch
)

val_loader = DataLoader(
    val_ds,
    batch_size=128,        # Larger batch OK for val (no gradient memory)
    shuffle=False,
    num_workers=4,
    pin_memory=True,
)

print(f"Train batches: {len(train_loader)}")
print(f"Val batches:   {len(val_loader)}")

# Inspect one batch
X_batch, y_batch = next(iter(train_loader))
print(f"Batch X: {X_batch.shape}, y: {y_batch.shape}")
Output
Train batches: 12
Val batches:   2
Batch X: torch.Size([64, 20]), y: torch.Size([64])

num_workers: Setting num_workers > 0 spawns background processes that load the next batch while the GPU trains on the current one. On Windows, always wrap DataLoader usage in if __name__ == '__main__': to avoid multiprocessing errors. Start with num_workers=4 and increase until training speed stops improving.

03 — Transforms & Augmentation

Preprocessing Inside the Pipeline

Python
from torchvision import transforms

# Standard image transform pipeline
train_transform = transforms.Compose([
    transforms.RandomHorizontalFlip(p=0.5),
    transforms.RandomRotation(degrees=15),
    transforms.ColorJitter(brightness=0.2, contrast=0.2),
    transforms.ToTensor(),              # PIL/numpy → float32 tensor [0,1]
    transforms.Normalize(
        mean=[0.485, 0.456, 0.406],   # ImageNet stats
        std=[0.229, 0.224, 0.225]
    ),
])

val_transform = transforms.Compose([
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.485, 0.456, 0.406],
                           std=[0.229, 0.224, 0.225]),
])
# NOTE: Augmentations like RandomFlip are ONLY applied to training data.
# Validation uses only Normalize/ToTensor for consistent evaluation.

Data Augmentation — Train Only: Never apply random transforms (flip, crop, colour jitter) to your validation or test set. Those sets must be deterministic. Apply augmentation only in train_transform, not val_transform.
