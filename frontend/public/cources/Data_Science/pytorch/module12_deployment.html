classDriftMonitor: """ Detectinputdistributiondriftbycomparinglivefeaturestats

9

toreference(training)stats. Usespopulationstabilityindex(PSI)asthedriftmetric. """ def__init__(self,reference_mean:torch.Tensor,reference_std:torch.Tensor, n_features:int,z_threshold:float=3.0): self.ref_mean =reference_mean self.ref_std =reference_std self.z_threshold=z_threshold defcheck(self,batch:torch.Tensor)->dict: """Checkifbatchfeatureshavedriftedfromreference.""" batch_mean=batch.mean(dim=0) batch_std =batch.std(dim=0)

#Z-score:howmanystddeviationsisthelivemeanfromreference? z_scores=(batch_mean-self.ref_mean).abs()/(self.ref_std+1e-6)

drifted_features=(z_scores>self.z_threshold).nonzero(as_tuple=True)[0].tolist()

return{ 'drift_detected': len(drifted_features)>0, 'drifted_features': drifted_features, 'max_z_score': z_scores.max().item(), 'z_scores': z_scores } #Simulatedriftdetection ref_mean=torch.zeros(10) ref_std =torch.ones(10) drift_m =DriftMonitor(ref_mean,ref_std,n_features=10) #Normalbatch—nodrift normal_batch=torch.randn(100,10) result=drift_m.check(normal_batch) print(f"Normalbatch: drift={result['drift_detected']} max_z={result['max_z_score']:.2f}") #Driftedbatch—feature3and7shifted drifted_batch=torch.randn(100,10) drifted_batch[:,3]+=5.0 #Bigshiftinfeature3 drifted_batch[:,7]+=4.0 #Shiftinfeature7
